合并两个有序数组

###

public  int[] merge(int[] arr1, int[] arr2) {
    int[] arr3 = new int[arr1.length + arr2.length];
    merge0(arr1, arr2, arr3);
    return arr3;
}

public  void merge0(int[] arr1, int[] arr2, int[] arr3) {
    int[] min, max;

    //用 arr1[0],arr2[0] 判断大小队列 left == min right == max
    //主要方便理解

    if (arr1[0] <= arr2[0]) {
        min = arr1;
        max = arr2;
    } else {
        min = arr2;
        max = arr1;
    }

    //创建三个指针用于控制插入
    int left = 0, right = 0, middle = 0;
    //判断当前在左边还是右边
    boolean isLeft = true,isLoop=true;
    //找出min数组大于max[0]的第一个元素
    while (middle != arr3.length) {
        if (isLeft) {
            for (; left < min.length;) {
                if (max.length!= right && min[left] > max[right]) {  //&&交换位置数组下标会越界
                    isLeft = false;
                    break;
                }
                arr3[middle] = min[left];
                middle++;
                left++;
            }
        } else {
            for (; right < max.length;) {
                if ( min.length != left &&min[left] < max[right]) { //&&交换位置数组下标会越界
                    isLeft = true;
                    break;
                }
                arr3[middle] = max[right];
                middle++;
                right++;
            }
        }
        //当左右任意一边赋值完了
        if(isLoop) {
            if(max.length == right ||  min.length == left)
            {
                isLeft =!isLeft;
                isLoop =false;
            }
        }
    }

}
